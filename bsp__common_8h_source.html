263{
264 uint32_t clock_hz = 0;
265 uint32_t fselcpu0 = R_SYSC_S->SCKCR2_b.FSELCPU0;
266 uint32_t fselcpu1 = R_SYSC_S->SCKCR2_b.FSELCPU1;
267
268 switch (clock)
269 {
270 case FSP_PRIV_CLOCK_CPU0:
271 {
272 clock_hz = g_bsp_system_clock_select[clock][R_SYSC_S->SCKCR2_b.DIVSELSUB] >> fselcpu0;
273 break;
274 }
275
276 case FSP_PRIV_CLOCK_CPU1:
277 {
278 clock_hz = g_bsp_system_clock_select[clock][R_SYSC_S->SCKCR2_b.DIVSELSUB] >> fselcpu1;
279 break;
280 }
281
282 /* These iclk and pclk cases are intentionally combined. */
283 case FSP_PRIV_CLOCK_ICLK:
284 case FSP_PRIV_CLOCK_PCLKH:
285 case FSP_PRIV_CLOCK_PCLKM:
