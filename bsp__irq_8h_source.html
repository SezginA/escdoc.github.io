60__STATIC_INLINE void R_FSP_IsrContextSet (IRQn_Type const irq, void * p_context)
61{
62 /* This provides access to the ISR context array defined in bsp_irq.c. This is an inline function instead of
63 * being part of bsp_irq.c for performance considerations because it is used in interrupt service routines. */
64 if (irq >= 0)
65 {
66 gp_renesas_isr_context[irq] = p_context;
67 }
68 else
69 {
70 gp_renesas_isr_context[(uint32_t) irq + BSP_VECTOR_NUM_OFFSET + BSP_ICU_VECTOR_MAX_ENTRIES] = p_context;
71 }
72}

82__STATIC_INLINE void R_BSP_IrqClearPending (IRQn_Type irq)
83{
84 GICD_Type * GICD;
85 GICR_SGI_PPI_Type * GICR_TARGET0_INTREG;
86
87 GICD = BSP_PRV_GICD_ADDRESS(BSP_CFG_CPU);
88 GICR_TARGET0_INTREG = BSP_PRV_GICR_TARGET0_INTREG_ADDRESS(BSP_CFG_CPU);
89
90 if (irq >= 0)
91 {
92 uint32_t _irq = (uint32_t) irq;
93 GICD->GICD_ICPENDR[(((uint32_t) irq) /
94 BSP_VECTOR_NUM_OFFSET)] = (uint32_t) (0x00000001UL << (_irq % BSP_VECTOR_NUM_OFFSET));
95 }
96 else
97 {
98 uint32_t _irq = ((uint32_t) irq + BSP_VECTOR_NUM_OFFSET);
99 GICR_TARGET0_INTREG->GICR_ICPENDR0 = (uint32_t) (0x00000001UL << _irq);
100 }
101}

113__STATIC_INLINE uint32_t R_BSP_IrqPendingGet (IRQn_Type irq)
114{
115 GICD_Type * GICD;
116 GICR_SGI_PPI_Type * GICR_TARGET0_INTREG;
117 uint32_t value = 0;
118
119 GICD = BSP_PRV_GICD_ADDRESS(BSP_CFG_CPU);
120 GICR_TARGET0_INTREG = BSP_PRV_GICR_TARGET0_INTREG_ADDRESS(BSP_CFG_CPU);
121
122 if (irq >= 0)
123 {
124 uint32_t _irq = (uint32_t) irq;
125 uint32_t shift = (_irq % BSP_VECTOR_NUM_OFFSET);
126 value = (GICD->GICD_ISPENDR[(((uint32_t) irq) / BSP_VECTOR_NUM_OFFSET)] & (uint32_t) (0x00000001UL << shift)) >>
127 shift;
128 }
129 else
130 {
131 uint32_t _irq = ((uint32_t) irq + BSP_VECTOR_NUM_OFFSET);
132 uint32_t shift = _irq;
133 value = (GICR_TARGET0_INTREG->GICR_ISPENDR0 & (uint32_t) (0x00000001UL << shift)) >> shift;
134 }
135
136 return value;
137}

148__STATIC_INLINE void R_BSP_IrqCfg (IRQn_Type const irq, uint32_t priority, void * p_context)
149{
150#if (52U == __CORTEX_R)
151 GICD_Type * GICD;
152 GICR_SGI_PPI_Type * GICR_TARGET0_INTREG;
153
154 GICD = BSP_PRV_GICD_ADDRESS(BSP_CFG_CPU);
155 GICR_TARGET0_INTREG = BSP_PRV_GICR_TARGET0_INTREG_ADDRESS(BSP_CFG_CPU);
156
157 if (irq >= 0)
158 {
159 uint32_t _irq = (uint32_t) irq;
160
161 /* Set the interrupt group to 1 (IRQ) */
162 GICD->GICD_IGROUPR[(((uint32_t) irq) /
163 BSP_VECTOR_NUM_OFFSET)] |= (uint32_t) (0x00000001UL << (_irq % BSP_VECTOR_NUM_OFFSET));
164
165 /* Set the interrupt priority */
166 GICD->GICD_IPRIORITYR[((uint32_t) irq) / 4UL] &= (uint32_t) (~(BSP_PRV_IRQ_CFG_MSK << (8UL * (_irq % 4UL))));
167 GICD->GICD_IPRIORITYR[((uint32_t) irq) / 4UL] |= (priority << (3UL + (8UL * (_irq % 4UL))));
168
169 /* Store the context. The context is recovered in the ISR. */
170 R_FSP_IsrContextSet(irq, p_context);
171 }
172 else
173 {
174 uint32_t _irq = ((uint32_t) irq + BSP_VECTOR_NUM_OFFSET);
175
176 /* Set the interrupt group to 1 (IRQ) */
177 GICR_TARGET0_INTREG->GICR_IGROUPR0 |= (uint32_t) (0x00000001UL << _irq);
178
179 /* Set the interrupt priority */
180 GICR_TARGET0_INTREG->GICR_IPRIORITYR[(((uint32_t) irq + BSP_VECTOR_NUM_OFFSET)) /
181 4UL] &= (uint32_t) (~(BSP_PRV_IRQ_CFG_MSK << (8UL * (_irq % 4UL))));
182 GICR_TARGET0_INTREG->GICR_IPRIORITYR[(((uint32_t) irq + BSP_VECTOR_NUM_OFFSET)) /
183 4UL] |= (priority << (3UL + (8UL * (_irq % 4UL))));
184
185 /* Store the context. The context is recovered in the ISR. */
186 R_FSP_IsrContextSet(irq, p_context);
187 }
188#endif
189}
