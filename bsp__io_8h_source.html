340__STATIC_INLINE uint32_t R_BSP_PinRead (bsp_io_port_pin_t pin)
341{
342 uint32_t aselp = ((uint32_t) ((R_PTADR->RSELP[pin >> 8]) >> (pin & BSP_IO_PRV_8BIT_MASK)) & 0x00000001UL);
343 uint32_t pin_level;
344
345 if (0U == aselp)
346 {
347 /* Read pin level. */
348 pin_level = (uint32_t) ((R_PORT_SR->PIN[pin >> 8]) >> (pin & BSP_IO_PRV_8BIT_MASK)) & 0x00000001UL;
349 }

351 {
352 /* Read pin level. */
353 pin_level = (uint32_t) ((R_PORT_NSR->PIN[pin >> 8]) >> (pin & BSP_IO_PRV_8BIT_MASK)) & 0x00000001UL;

365__STATIC_INLINE void R_BSP_PinWrite (bsp_io_port_pin_t pin, bsp_io_level_t level)
366{
367 /* Set output level and pin direction to output. */
368 uint32_t port_num = (uint32_t) (pin >> 8);
369 uint32_t pin_bit = (uint32_t) (pin & BSP_IO_PRV_8BIT_MASK);
370 uint32_t lvl = (uint32_t) level;
371 uint32_t aselp = ((uint32_t) ((R_PTADR->RSELP[pin >> 8]) >> pin_bit) & 0x00000001UL);
372
373 if (0U == aselp)
374 {
375 R_PORT_SR->PM[port_num] = (uint16_t) ((R_PORT_SR->PM[port_num]) | (BSP_IO_PM_OUTPUT << (2UL * pin_bit)));
376 R_PORT_SR->P[port_num] = (uint8_t) (((R_PORT_SR->P[port_num]) & (~(1UL << pin_bit))) | (lvl << pin_bit));
377 }
378 else
379 {
380 R_PORT_NSR->PM[port_num] = (uint16_t) ((R_PORT_NSR->PM[port_num]) | (BSP_IO_PM_OUTPUT << (2UL * pin_bit)));
381 R_PORT_NSR->P[port_num] = (uint8_t) (((R_PORT_NSR->P[port_num]) & (~(1UL << pin_bit))) | (lvl << pin_bit));
382 }
383}
